import “ADAppDelegate.h”

 
@implementation ADAppDelegate
@synthesize window = _window;
@synthesize allPlaces, motionState;

/*Awesome!*/
- (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions
{
    // Override point for customization after application launch.
    [Alohar startSession:@"APIKey" withDelegate:self];
    [Alohar startUpdatingMotionState];
    [Alohar startUpdatingUserStays];
    return YES;
}

- (void)didUpdateToMotionState:(ALMotionState *)newMotionState fromMotionState:(ALMotionState *)oldMotionState
{
    NSLog(@"New Motion State: %d", [newMotionState.state intValue]);

    self.motionState = newMotionState;
}
- (void)didUpdateToUserStay:(ALUserStay *)newStay fromUserStay:(ALUserStay *)oldStay
{
    NSLog(@"New Motion State: %@", newStay);
}

- (void)placesRequestFinished:(ALResponse *)response
{
    NSLog(@"places request completed: %d total places", response.objects.count);
    self.allPlaces = response.objects;
}

- (void)aloharDidFailWithError:(NSError *)error
{
    NSLog(@"Sorry, there was an error with alohar: %@", error);
}
